Code Example 1: Solving Equations

# Example Python code to solve quadratic equations
import math

def solve_quadratic(a, b, c):
    discriminant = b**2 - 4*a*c
    if discriminant > 0:
        root1 = (-b + math.sqrt(discriminant)) / (2*a)
        root2 = (-b - math.sqrt(discriminant)) / (2*a)
        return root1, root2
    elif discriminant == 0:
        root = -b / (2*a)
        return root,
    else:
        return "No real roots"

# Example usage
print(solve_quadratic(1, -3, 2))  # Output: (2.0, 1.0)
