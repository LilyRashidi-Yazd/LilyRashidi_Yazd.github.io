Code Example 1: Solving Kinematic Equations

# Python code to solve kinematic equations
# Given initial velocity (v0), acceleration (a), and time (t), we calculate final velocity (v)

def final_velocity(v0, a, t):
    return v0 + a * t

# Example usage
initial_velocity = 10  # m/s
acceleration = 2       # m/s^2
time = 5               # seconds

print(final_velocity(initial_velocity, acceleration, time))  # Output: 20 m/s

Code Example 2: Projectile Motion

# Python code to simulate projectile motion
import math

def projectile_motion(initial_velocity, angle):
    g = 9.81  # acceleration due to gravity (m/s^2)
    angle_rad = math.radians(angle)  # Convert angle to radians
    v_x = initial_velocity * math.cos(angle_rad)
    v_y = initial_velocity * math.sin(angle_rad)
    
    time_of_flight = 2 * v_y / g
    max_height = (v_y ** 2) / (2 * g)
    range_of_projectile = v_x * time_of_flight
    
    return time_of_flight, max_height, range_of_projectile

# Example usage
velocity = 20  # m/s
angle = 45     # degrees

time, height, range_ = projectile_motion(velocity, angle)
print(f"Time of flight: {time} s")
print(f"Max height: {height} m")
print(f"Range: {range_} m")
